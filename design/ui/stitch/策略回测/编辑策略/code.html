import pandas as pd
from gzt_api import Strategy, Order

class ValueStrategy(Strategy):
    def on_init(self):
        # 初始化沪深300权重股池
        self.universe = ['000001.SH', '600519.SH']
        self.set_benchmark('000300.SH')

    def on_bar(self, bar):
        # 简单的均线交叉逻辑
        ma5 = bar.close.rolling(5).mean()
        ma20 = bar.close.rolling(20).mean()

        if ma5 > ma20 and self.pos == 0:
            self.buy(bar.symbol, bar.close, 1000)
        elif ma5 < ma20 and self.pos > 0:
            self.sell(bar.symbol, bar.close, self.pos)
